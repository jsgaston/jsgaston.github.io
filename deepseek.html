Configuración de Google Apps Script

Para usar esta aplicación, necesitas configurar un script de Google Apps que actuará como intermediario con Google Drive.

Paso 1: Crear un nuevo proyecto en Google Apps Script

  1. Ve a script.google.com
  2. Crea un nuevo proyecto
  3. Pega el siguiente código:
// Código para Google Apps Script function doPost(e) { try { var data = JSON.parse(e.postData.contents); var action = data.action; if (action === "upload") { return uploadFile(data); } else if (action === "list") { return listFiles(data); } else if (action === "download") { return downloadFile(data); } else { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Acción no válida" })).setMimeType(ContentService.MimeType.JSON); } } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Solo se permiten solicitudes POST" })).setMimeType(ContentService.MimeType.JSON); } function uploadFile(data) { var folderId = "TU_FOLDER_ID"; // Reemplaza con el ID de tu carpeta en Drive var folder = DriveApp.getFolderById(folderId); var fileName = data.fileName; var fileType = data.fileType; var content = data.content; var file; if (fileType === "text") { file = folder.createFile(fileName + ".txt", content, "text/plain"); } else if (fileType === "image") { // Eliminar el prefijo de datos de la imagen (data:image/png;base64,) var base64Data = content.replace(/^data:image\/\w+;base64,/, ""); var blob = Utilities.newBlob(Utilities.base64Decode(base64Data), "image/png", fileName + ".png"); file = folder.createFile(blob); } else { throw new Error("Tipo de archivo no válido"); } return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "Archivo subido correctamente", fileId: file.getId(), fileUrl: file.getUrl() })).setMimeType(ContentService.MimeType.JSON); } function listFiles(data) { var folderId = "TU_FOLDER_ID"; // Reemplaza con el ID de tu carpeta en Drive var folder = DriveApp.getFolderById(folderId); var files = folder.getFiles(); var fileType = data.fileType || "all"; var result = []; while (files.hasNext()) { var file = files.next(); var name = file.getName(); var id = file.getId(); var url = file.getUrl(); var type = name.endsWith(".txt") ? "text" : "image"; if (fileType === "all" || fileType === type) { result.push({ id: id, name: name, url: url, type: type }); } } return ContentService.createTextOutput(JSON.stringify({ status: "success", files: result })).setMimeType(ContentService.MimeType.JSON); } function downloadFile(data) { var fileId = data.fileId; var file = DriveApp.getFileById(fileId); if (file.getMimeType() === "text/plain") { var content = file.getBlob().getDataAsString(); return ContentService.createTextOutput(JSON.stringify({ status: "success", content: content, type: "text" })).setMimeType(ContentService.MimeType.JSON); } else { var blob = file.getBlob(); var base64Data = Utilities.base64Encode(blob.getBytes()); var mimeType = blob.getContentType(); var dataUrl = "data:" + mimeType + ";base64," + base64Data; return ContentService.createTextOutput(JSON.stringify({ status: "success", content: dataUrl, type: "image" })).setMimeType(ContentService.MimeType.JSON); } }

Paso 2: Configurar el script

  1. Reemplaza "TU_FOLDER_ID" con el ID de la carpeta de Google Drive donde quieres guardar los archivos
  2. Guarda el proyecto
  3. Despliega el script como una aplicación web
  4. Configura el acceso para "Anyone, even anonymous" (si quieres acceso público)
  5. Copia la URL de despliegue

Paso 3: Configurar la aplicación

Instrucciones de Uso

  1. Configura el script de Google Apps Script siguiendo las instrucciones en la pestaña "Configuración"
  2. En la pestaña "Aplicación", selecciona el tipo de archivo que quieres subir (texto o imagen)
  3. Proporciona un nombre para el archivo
  4. Haz clic en "Subir a Google Drive" para guardar el archivo
  5. Usa el botón "Cargar Archivos" para ver y gestionar los archivos en tu Drive
